public static void main(String[] args) {
		for(int a=0;a<=6;a++) {
			for(int b=5;b>=a;b--) {
				System.out.print(" ");
			}
			for(int c=1;c<=(2*a-1);c++) {
				System.out.print("*");
			}
			System.out.println();
			}
		for(int a=5;a>=0;a--) {
			for(int b=5;b>=a;b--) {
				System.out.print(" ");
			}
			for(int c=1;c<=(2*a-1);c++) {
				System.out.print("*");
			}
			System.out.println();
			}
	}
